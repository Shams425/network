:foreach u in=[/ip hotspot user find] do={

    :local uptime [/ip hotspot user get $u uptime];
    :local limitUptime [/ip hotspot user get $u limit-uptime];

    :local bytesIn [/ip hotspot user get $u bytes-in];
    :local bytesOut [/ip hotspot user get $u bytes-out];

    :local limitBytes [/ip hotspot user get $u limit-bytes-total];

    :local totalBytes ($bytesIn + $bytesOut);

    :local remove false;

    :if (($limitUptime != "") && ($uptime >= $limitUptime)) do={
        :set remove true;
    };

    :if (($limitBytes != "") && ($limitBytes > 0) && ($totalBytes >= $limitBytes)) do={
        :set remove true;
    };

    :if ($remove = true) do={

        :local uname [/ip hotspot user get $u name];

        /ip hotspot active remove [find user=$uname];

        /ip hotspot user remove $u;

    };
};
